See cost, latency, privacy, repeats, and provider performance first. Compile the repeated work later.

# Drop-in capture
export OPENAI_BASE_URL=http://127.0.0.1:8787/v1
export OPENAI_API_KEY=kolm_local_dev

# Your app keeps using its normal client.
# kolm stores every approved row locally.
# Promote the repeat
kolm datasets build support-triage --from lake
kolm train support-triage --teacher openrouter:auto
kolm compile support-triage --target gemma-3n
kolm verify support-triage.kolm
