Install

The JS SDK is a single ES module served from https://kolm.ai/sdk.js. Python, Rust, and Go call the HTTP API directly with their language's standard HTTP client.

// Browser, Deno, Bun, Cloudflare Workers, Node 22+. No install step.
import { recipe, Recipe } from 'https://kolm.ai/sdk.js';

// Or pin a version (recommended for production):
import { recipe } from 'https://kolm.ai/sdk@1.js';
# Python 3.10+. Stdlib only, no install required for the HTTP path.
import json, urllib.request, urllib.error

# The Python wrapper ships with the CLI install (vendored helpers).
# See /docs/cli for the install path; no separate Python package today.
from recipe import RecipeClient
# Cargo.toml
[dependencies]
reqwest = { version = "0.12", features = ["json", "blocking"] }
serde = { version = "1", features = ["derive"] }
serde_json = "1"
// Go 1.21+. Stdlib only, no module install required.
import (
 "bytes"
 "encoding/json"
 "net/http"
)

Authentication

All authenticated routes accept a bearer token via Authorization: Bearer ks_.... Public registry, health, and signup routes do not require a key. Keys can be issued from /account.

import { Recipe } from 'https://kolm.ai/sdk.js';

const r = new Recipe({
 apiKey: process.env.KOLM_KEY, // 'ks_live_...'
 base: 'https://kolm.ai',
});
import os, urllib.request

KEY = os.environ['KOLM_KEY']
BASE = 'https://kolm.ai'

def auth_request(path, method='GET', body=None):
 data = json.dumps(body).encode() if body else None
 req = urllib.request.Request(BASE + path, data=data, method=method)
 req.add_header('Authorization', 'Bearer ' + KEY)
 if data: req.add_header('Content-Type', 'application/json')
 return req

Core operations

Every public method on the JS SDK is documented below with the equivalent HTTP request in Python, Rust, and Go.

new Recipe(opts) construct a client bound to a base URL and API key

The constructor accepts {base, apiKey, cache, unsafeMode, timeoutMs}. None are required. Defaults: base resolves from the import URL origin, cache uses localStorage, recipes execute inside a Web Worker sandbox, and the per-recipe timeout is 1000ms.

const r = new Recipe({
 base: 'https://kolm.ai',
 apiKey: 'ks_live_...',
 cache: true,
 unsafeMode: false,
 timeoutMs: 1000,
});

load() hydrate the public recipe registry once

Calls GET /v1/registry/export and caches the response in localStorage under recipe.registry.v1. Subsequent calls return the cached registry. Each recipe entry holds its source, source hash, and version id.

run(name, input) execute one recipe on this device, return signed receipt

Runs the named recipe inside a Web Worker sandbox by default (no window, fetch, or localStorage access). Falls back to in-page compile only on runtimes without Worker support. Records the run in recipe.runs. The HTTP equivalent is POST /v1/run.

Returns a Proxy over your existing client (Anthropic, OpenAI, etc.). When opts.verified is set, client.messages.create(...) is intercepted and routed to /v1/wrap/verified. Error handling

kolm.ai returns standard HTTP status codes. The common failure cases are 401 (missing or rotated key), 429 (rate limited, retry with backoff), 503 (upstream model or verifier briefly unavailable), and plain network errors. The pattern below catches each cleanly in every language.

async function withRetry(fn, max = 3) {
 let attempt = 0, lastErr;
 while (attempt++ < max) {
 try {
 return await fn();
 } catch (e) {
 lastErr = e;
 const status = e.status || 0;
 if (status === 429 || status === 503) {
 const wait = Math.min(1000 * 2 ** attempt, 8000);
 await new Promise(r => setTimeout(r, wait));
 continue;
 }
 if (status === 401) throw new Error('rotate your KOLM_KEY');
 throw e; // network or 4xx: do not retry
 }
 }
 throw lastErr;
}

const out = await withRetry(() => recipe.verified({ /* ... */ }));
import time, json, urllib.request, urllib.error

def with_retry(fn, max_attempts=3):
 for attempt in range(1, max_attempts + 1):
 try:
 return fn()
 except urllib.error.HTTPError as e:
 if e.code == 401:
 raise RuntimeError('rotate your KOLM_KEY') from None
 if e.code in (429, 503):
 wait = min(2 ** attempt, 8)
 time.sleep(wait)
 continue
 raise
 except urllib.error.URLError:
 if attempt == max_attempts:
 raise
 time.sleep(2 ** attempt)
 raise RuntimeError('exhausted retries')
use std::{thread, time::Duration};

fn with_retry<F, T>(mut fn_: F, max: u32) -> Result<T, reqwest::Error>
where F: FnMut() -> Result<reqwest::blocking::Response, reqwest::Error> + Sized,
 T: serde::de::DeserializeOwned,
{
 let mut attempt = 0;
 loop {
 attempt += 1;
 let r = fn_()?;
 let status = r.status().as_u16();
 if status == 401 { panic!("rotate KOLM_KEY"); }
 if (status == 429 || status == 503) && attempt < max {
 thread::sleep(Duration::from_secs(1 << attempt.min(3)));
 continue;
 }
 return r.json();
 }
}
func withRetry(c *http.Client, req *http.Request, max int) (*http.Response, error) {
 var last error
 for attempt := 1; attempt <= max; attempt++ {
 resp, err := c.Do(req)
 if err != nil {
 last = err
 time.Sleep(time.Duration(1<<attempt) * time.Second)
 continue
 }
 if resp.StatusCode == 401 {
 return nil, fmt.Errorf("rotate KOLM_KEY")
 }
 if resp.StatusCode == 429 || resp.StatusCode == 503 {
 resp.Body.Close()
 time.Sleep(time.Duration(1<<attempt) * time.Second)
 continue
 }
 return resp, nil
 }
 return nil, last
}
